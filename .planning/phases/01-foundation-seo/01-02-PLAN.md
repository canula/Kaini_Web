**4. Vanilla JS Hamburger Toggle (INFR-02):**

Place a `<script>` block at the end of `<body>` (before `</body>`):

```html
<script>
  const navToggle = document.getElementById('nav-toggle');
  const mobileMenuDropdown = document.getElementById('mobile-menu-dropdown');

  navToggle.addEventListener('click', () => {
    const expanded = navToggle.getAttribute('aria-expanded') === 'true';
    navToggle.setAttribute('aria-expanded', String(!expanded));
    mobileMenuDropdown.classList.toggle('hidden');
  });

  // Close menu on Escape key
  document.addEventListener('keydown', (e) => {
    if (e.key === 'Escape') {
      navToggle.setAttribute('aria-expanded', 'false');
      mobileMenuDropdown.classList.add('hidden');
    }
  });

  // Close menu when a nav link is clicked (smooth scroll then close)
  mobileMenuDropdown.querySelectorAll('a').forEach((link) => {
    link.addEventListener('click', () => {
      navToggle.setAttribute('aria-expanded', 'false');
      mobileMenuDropdown.classList.add('hidden');
    });
  });
</script>
```

Must use vanilla JS only — no frameworks, no CSS-only checkbox hack (user explicitly rejected it).
